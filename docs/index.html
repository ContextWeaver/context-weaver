Basic Usage
Advanced
const { RPGEventGenerator } = require('rpg-event-generator'); const generator = new RPGEventGenerator(); // Generate a simple event const event = generator.generateEvent(); console.log(event.title); console.log(event.description);
const { RPGEventGenerator } = require('rpg-event-generator'); const generator = new RPGEventGenerator({ theme: 'fantasy', difficulty: 'medium', context: { playerLevel: 5, location: 'forest', season: 'spring', timeOfDay: 'day' } }); // Generate context-aware event const event = generator.generateEvent(); console.log(`Generated: ${event.title}`); console.log(`Difficulty: ${event.difficulty}`); console.log(`Choices: ${event.choices.length}`);
